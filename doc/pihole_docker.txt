# controle de memoria e cpu
$ sudo vi /boot/firmware/cmdline.txt
(...) cgroup_enable=cpuset cgroup_memory=1 cgroup_enable=memory
$ reboot

# Docker Compose
$ vi docker-compose.yml
services:
  pihole:
    container_name: pihole
    image: pihole/pihole:latest
    hostname: pihole
    deploy:
      resources:
        limits:
          cpus: '0.50'
          memory: 500M
    network_mode: bridge
    dns:
      - 192.168.1.10
    ports:
      - "192.168.1.12:53:53/udp"
      - "8080:80/tcp"
    environment:
      TZ: 'America/Sao_Paulo'
      WEBPASSWORD: 'password'
      DNSMASQ_LISTENING: 'all'
      TEMPERATUREUNIT: 'c'
    volumes:
      - './etc-pihole:/etc/pihole'
      - './etc-dnsmasq.d:/etc/dnsmasq.d'
    cap_add:
      - NET_ADMIN
    restart: unless-stopped

# Ativando o Pihole
$ docker compose up -d

# Alterando senha de login no Pi-Hole
$ docker exec -it pihole /bin/bash
$ pihole -a -p somepasswordhere

# atualização do pihole
$ docker pull pihole/pihole && docker rm -f pihole && docker-compose -f pihole/docker-compose.yml up -d && docker system prune --all --force
